`timetableUtils.ts` を以下に置き換える:

```typescript
import type { DirectionSchedule } from '../data/timetable'

export type DayType = 'weekday' | 'saturday' | 'holiday'

/** "HH:MM" を深夜0時からの分数に変換 */
export function toMinutes(time: string): number {
  const [h, m] = time.split(':').map(Number)
  return h * 60 + m
}

/**
 * 指定日の種別を返す（平日 / 土曜 / 休日）
 * ⚠️ 日本の祝日は考慮しない（土・日のみ判定）
 */
export function getDayType(date: Date = new Date()): DayType {
  const day = date.getDay()
  if (day === 6) return 'saturday'
  if (day === 0) return 'holiday'
  return 'weekday'
}

/**
 * DayType に応じて適切なダイヤを返す
 * saturday フィールドが未定義の場合は holiday にフォールバック
 */
export function getSchedule(dir: DirectionSchedule, dayType: DayType): string[] {
  if (dayType === 'saturday') return dir.saturday ?? dir.holiday
  if (dayType === 'holiday') return dir.holiday
  return dir.weekday
}

/** 現在時刻以降の発車時刻のみ返す */
export function filterUpcoming(departures: string[], currentTime: string): string[] {
  const currentMinutes = toMinutes(currentTime)
  return departures.filter((t) => toMinutes(t) >= currentMinutes)
}
```

```typescript
'use client'

import { useEffect, useMemo, useState } from 'react'
import { timetable, type TabGroup, type RouteData } from '../data/timetable'
import { getDayType, getSchedule, filterUpcoming } from '../lib/timetableUtils'

const basePath = process.env.NEXT_PUBLIC_BASE_PATH ?? ''

const ROUTE_LOGOS: Record<string, string> = {
  'りんかい線': `${basePath}/transport/rinkai.svg`,
  'ゆりかもめ': `${basePath}/transport/yurikamome.svg`,
  '都バス': `${basePath}/transport/toei.svg`,
  'BRT': `${basePath}/transport/brt.png`,
}

const TABS: { key: TabGroup; label: string }[] = [
  { key: 'rail', label: '鉄道' },
  { key: 'brt', label: 'BRT' },
  { key: 'bus', label: 'バス' },
]

function getNowString(): string {
  const now = new Date()
  return `${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}`
}

type RouteCardProps = {
  route: RouteData & { directions: Array<{ label: string; upcoming: string[] }> }
}

function RouteCard({ route }: RouteCardProps) {
  const logo = ROUTE_LOGOS[route.name]
  const maxRows = Math.max(0, ...route.directions.map((d) => d.upcoming.length))
  const isEmpty = maxRows === 0

  return (
    <div className="rounded-xl border border-slate-200 bg-white overflow-hidden shadow-sm">
      {/* ヘッダー */}
      <div className="flex items-center gap-3 px-4 py-3 bg-slate-50 border-b border-slate-200">
        {logo && (
          <img src={logo} alt={route.name} className="h-6 object-contain flex-shrink-0" />
        )}
        <div>
          <p className="font-semibold text-slate-800 text-sm">{route.name}</p>
          <p className="text-xs text-slate-500">{route.station}　徒歩{route.walkMinutes}分</p>
        </div>
      </div>

      {isEmpty ? (
        <p className="px-4 py-4 text-sm text-slate-400 text-center">本日の終電は終了しました</p>
      ) : (
        <div>
          {/* 方面ヘッダー */}
          <div className={`grid grid-cols-${route.directions.length} divide-x divide-slate-200 border-b border-slate-100`}>
            {route.directions.map((dir) => (
              <div key={dir.label} className="px-3 py-1.5 text-xs font-medium text-slate-600 text-center bg-slate-50">
                {dir.label}
              </div>
            ))}
          </div>
          {/* 時刻行 */}
          {Array.from({ length: Math.min(maxRows, 6) }).map((_, i) => (
            <div
              key={i}
              className={`grid grid-cols-${route.directions.length} divide-x divide-slate-100 ${i % 2 === 0 ? 'bg-white' : 'bg-slate-50/50'}`}
            >
              {route.directions.map((dir) => (
                <div key={dir.label} className="px-3 py-1.5 text-sm tabular-nums text-center text-slate-700">
                  {dir.upcoming[i] ?? ''}
                </div>
              ))}
            </div>
          ))}
          {maxRows > 6 && (
            <p className="px-4 py-1.5 text-xs text-slate-400 text-center">
              以降 {maxRows - 6} 本
            </p>
          )}
        </div>
      )}
    </div>
  )
}

export default function TransportView() {
  const [activeTab, setActiveTab] = useState<TabGroup>('rail')
  const [now, setNow] = useState(getNowString)
  const dayType = getDayType()

  useEffect(() => {
    const timer = setInterval(() => setNow(getNowString()), 60_000)
    return () => clearInterval(timer)
  }, [])

  const routes = useMemo(
    () =>
      timetable
        .filter((r) => r.tabGroup === activeTab)
        .map((route) => ({
          ...route,
          directions: route.directions.map((dir) => ({
            ...dir,
            upcoming: filterUpcoming(getSchedule(dir, dayType), now),
          })),
        })),
    [activeTab, now, dayType]
  )

  const dayLabel =
    dayType === 'saturday' ? '土曜ダイヤ' : dayType === 'holiday' ? '休日ダイヤ' : '平日ダイヤ'

  return (
    <div className="flex flex-col h-full">
      {/* 現在時刻・ダイヤ種別 */}
      <p className="px-4 pt-3 pb-1 text-xs text-slate-400">
        {now} 現在 · {dayLabel}
      </p>

      {/* タブバー */}
      <div className="flex border-b border-slate-200 px-4" role="tablist">
        {TABS.map(({ key, label }) => (
          <button
            key={key}
            role="tab"
            aria-selected={activeTab === key}
            onClick={() => setActiveTab(key)}
            className={`px-5 py-2.5 text-sm font-medium border-b-2 transition-colors ${
              activeTab === key
                ? 'border-blue-500 text-blue-600'
                : 'border-transparent text-slate-500 hover:text-slate-700'
            }`}
          >
            {label}
          </button>
        ))}
      </div>

      {/* ルートカード */}
      <div className="flex-1 overflow-y-auto px-4 py-4 space-y-4">
        {routes.map((route) => (
          <RouteCard key={`${route.name}-${route.station}`} route={route} />
        ))}
      </div>
    </div>
  )
}
```
